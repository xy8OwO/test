void sw_spi_write_byte(uint8_t data)
{
	uint8_t i;

	SPI_CS_CLEAR();
	for(i=0;i<8;i++)
	{			  
		SPI_SCK_CLEAR();
		if(data&0x80)
		{
		   SPI_MOSI_SET();
		}
		else
		{
		   SPI_MOSI_CLEAR();
		}
		SPI_SCK_SET();
		dat<<=1;
	}	
	SPI_CS_SET();	
}
